2

Supabase — database schema (SQL)

Copy and run this in your Supabase SQL Editor to create all required tables.

-- Churches table
create table churches (
  id bigint primary key generated always as identity,
  name text not null,
  faith text, denom text, founded int,
  address text, lat float, lng float,
  age_group text, service_time text, language text,
  website text, events text,
  wheelchair bool default false,
  kids bool default false,
  featured bool default false,
  claimed bool default false,
  rating float default 0,
  created_at timestamptz default now()
);

-- Reviews table
create table reviews (
  id bigint primary key generated always as identity,
  church_id bigint references churches(id),
  user_id uuid references auth.users(id),
  user_name text, stars int, body text,
  created_at timestamptz default now()
);

-- Claims table
create table claims (
  id bigint primary key generated always as identity,
  church_id bigint references churches(id),
  name text, email text, role text,
  plan text default 'free',
  status text default 'pending',
  created_at timestamptz default now()
);

-- Enable Row Level Security
alter table churches enable row level security;
alter table reviews enable row level security;
create policy "public read" on churches for select using (true);
create policy "public read" on reviews for select using (true);

Push your code to GitHub, then import the repo in Vercel. Add these environment variables in the Vercel dashboard under Settings → Environment Variables:

GOOGLE_MAPS_KEY=AIzaSy...

  ?query=churches+in+New+York
  &type=church
  &key=YOUR_API_KEY

// Then insert each result into Supabase:
await supabase.from('churches').insert(places.map(p => ({
